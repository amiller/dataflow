h1. What's this?

This is a Ruby implementation of Erlang style actors built on top of larrytheliquid's dataflow variables.

h1. Examples


<pre>
Ping = ErlangProcess.new {
  3.times {
    msg = receive
    case msg
    when "Ping"
      puts "Ping"
      Pong.send "Pong"
    end
  }
}

Pong = ErlangProcess.new {
  3.times {
    msg = receive
    case msg
    when "Pong"
      puts "Pong"

    end
  }
}

ErlangProcess.new {
  send "Hi"

}

sleep 1
